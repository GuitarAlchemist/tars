```fsharp
open System

let calculateAverage () =
    try
        let filePath = "sample.csv"
        let content = File.ReadAllText(filePath)
        let lines = content.Split('\n')
        let values = 
            lines 
            |> Array.filter (fun line -> line.Trim().Length > 0)
            |> Array.map (fun line ->
                let parts = line.Split(',')
                match parts |> Array.tryItem 1 with
                | Some value -> 
                    match Double.TryParse(value) with
                    | (true, num) -> Some num
                    | _ -> None
                | None -> None)
            |> Array.choose id

        if values.Length = 0 then
            "Error: No valid data to compute average."
        else
            let average = values |> Array.average
            sprintf "Average: %.2f" average
    with
    | :? FileNotFoundException -> 
        "Error: File not found. Ensure 'sample.csv' exists."
    | :? FormatException -> 
        "Error: Invalid numeric format in CSV data."
    | ex -> 
        $"Error: {ex.Message}"
```

   - Catches `FileNotFoundException` for missing files.
