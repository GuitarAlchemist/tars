```cpp
// CUDA: generate_prime_triplets.cu

// This kernel identifies prime triplets of the form (p, p+2, p+6)
// and stores the base prime `p` in output array if the pattern holds.

#include <math.h>
#include <stdio.h>

__device__ bool isPrime(int n) {
    if (n < 2) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    int limit = (int)sqrtf((float)n) + 1;
    for (int i = 3; i <= limit; i += 2) {
        if (n % i == 0) return false;
    }
    return true;
}

__global__ void generatePrimeTriplets(int limit, int* output, int* count) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= limit) return;

    if (isPrime(idx) && isPrime(idx + 2) && isPrime(idx + 6)) {
        int i = atomicAdd(count, 1);
        output[i * 3 + 0] = idx;
        output[i * 3 + 1] = idx + 2;
        output[i * 3 + 2] = idx + 6;
    }
}

// Host-side wrapper for testing
extern "C" void runPrimeTripletKernel(int limit) {
    int* d_output;
    int* d_count;
    int maxTriplets = 10000; // reserve enough space

    cudaMalloc(&d_output, sizeof(int) * maxTriplets * 3);
    cudaMalloc(&d_count, sizeof(int));
    cudaMemset(d_count, 0, sizeof(int));

    int blockSize = 256;
    int numBlocks = (limit + blockSize - 1) / blockSize;
    generatePrimeTriplets<<<numBlocks, blockSize>>>(limit, d_output, d_count);

    cudaDeviceSynchronize();

    // Optional: copy and print results
    int count;
    cudaMemcpy(&count, d_count, sizeof(int), cudaMemcpyDeviceToHost);
    int* h_output = (int*)malloc(sizeof(int) * count * 3);
    cudaMemcpy(h_output, d_output, sizeof(int) * count * 3, cudaMemcpyDeviceToHost);

    for (int i = 0; i < count; ++i) {
        printf("Triplet %d: (%d, %d, %d)\n", i, h_output[i*3], h_output[i*3+1], h_output[i*3+2]);
    }

    cudaFree(d_output);
    cudaFree(d_count);
    free(h_output);
}
```

```
// ======================================
// 🧠 TARS .trsx Diff Engine (Version Trace)
// ======================================
// Compares two .trsx files and outputs a semantic diff

module Tars.TrsxDiff

open System
open System.IO
open System.Text.RegularExpressions

/// Load a .trsx file as raw sections
let loadSections (path: string) : Map<string, string> =
    let content = File.ReadAllText(path)
    let sectionPattern = Regex("(?<hdr>^[a-z_]+)\s*{(?<body>.*?)^}\s*", RegexOptions.Singleline ||| RegexOptions.Multiline)
    sectionPattern.Matches(content)
    |> Seq.cast<Match>
    |> Seq.map (fun m -> m.Groups.["hdr"].Value.Trim(), m.Groups.["body"].Value.Trim())
    |> Map.ofSeq

/// Compute line-level diffs between two section bodies
let diffLines (label: string) (left: string) (right: string) =
    let lset = Set.ofSeq (left.Split('\n'))
    let rset = Set.ofSeq (right.Split('\n'))
    let added = Set.difference rset lset
    let removed = Set.difference lset rset
    (label, added, removed)

/// Generate a semantic diff report
let diffTrsxFiles (oldPath: string) (newPath: string) =
    let oldSections = loadSections oldPath
    let newSections = loadSections newPath

    let allKeys = Set.union (oldSections |> Map.keySet) (newSections |> Map.keySet)
    let diffs =
        allKeys
        |> Seq.map (fun key ->
            let oldVal = Map.tryFind key oldSections |> Option.defaultValue ""
            let newVal = Map.tryFind key newSections |> Option.defaultValue ""
            diffLines key oldVal newVal)
        |> Seq.toList

    // Output to console
    for (label, added, removed) in diffs do
        if not added.IsEmpty || not removed.IsEmpty then
            printfn "# Diff in section: %s" label
            if not added.IsEmpty then
                printfn "+ Added lines:\n%s" (String.Join("\n", added))
            if not removed.IsEmpty then
                printfn "- Removed lines:\n%s" (String.Join("\n", removed))

/// Example usage
// diffTrsxFiles "prime_triplet_001.trsx" "prime_triplet_002.trsx"

```

```cpp
    // Simple threshold split at midpoint value (assumes sorted externally or precomputed threshold)
```

```
open System

/// A Hurwitz quaternion is (a + bi + cj + dk) where a,b,c,d ∈ ℤ or ℤ+½
/// We'll represent them as (a,b,c,d,isHalf) where isHalf = true means all components are offset by ½

type HurwitzQuat = {
    A: float
    B: float
    C: float
    D: float
    IsHalf: bool // if true, components are a+½, b+½, etc.
}

/// Compute norm squared: N(q) = a² + b² + c² + d²
let norm2 (q: HurwitzQuat) =
    let offset = if q.IsHalf then 0.5 else 0.0
    let a = q.A + offset
    let b = q.B + offset
    let c = q.C + offset
    let d = q.D + offset
    a*a + b*b + c*c + d*d

/// Check if a given number is prime (simple test)
let isPrime (n: int) =
    if n <= 1 then false
    elif n = 2 then true
    elif n % 2 = 0 then false
    else
        let sqrtN = int (sqrt (float n))
        [3..2..sqrtN] |> List.forall (fun i -> n % i <> 0)

/// Generate Hurwitz quaternions within given bound and filter prime norms
let generateHurwitzPrimes (bound: int) =
    let mutable results = []

        for b in -bound..bound do
```

```
    let a = node.FeedbackScore |> int
    let b = node.Beliefs.Length
    let c = node.TaskIds.Length

    let isHalf = (a + b + c + d) % 2 = 1

        B = float b
        C = float c
        D = float d
        IsHalf = isHalf
    }

    let dx = a.A - b.A
    let dy = a.B - b.B
    let dz = a.C - b.C
    let dw = a.D - b.D
    sqrt (dx*dx + dy*dy + dz*dz + dw*dw)

/// Detects if the evolution between two versions is Hurwitz-prime in norm
```

```
// 🧬 TARS QuaternionCrossover: Belief-Aligned Semantic Recombination
// =====================================================================
// Combines two agent .trsx files if their Hurwitz quaternions are aligned

module Tars.AgentOps.QuaternionCrossover

open System
open System.IO
open Tars.TrsxGraph
open Tars.Math.HurwitzEvolution

/// Compute dot product of two normalized Hurwitz quaternions (as vectors)
let quaternionDot (q1: HurwitzQuat) (q2: HurwitzQuat) =
    let a1, b1, c1, d1 = q1.A, q1.B, q1.C, q1.D
    let a2, b2, c2, d2 = q2.A, q2.B, q2.C, q2.D
    let n1 = sqrt (a1*a1 + b1*b1 + c1*c1 + d1*d1)
    let n2 = sqrt (a2*a2 + b2*b2 + c2*c2 + d2*d2)
    ((a1*a2 + b1*b2 + c1*c2 + d1*d2) / (n1 * n2))

/// Return angle in degrees between two Hurwitz quaternions
let quaternionAngleDeg (q1: HurwitzQuat) (q2: HurwitzQuat) =
    acos (Math.Clamp(quaternionDot q1 q2, -1.0, 1.0)) * 180.0 / Math.PI

/// Merge beliefs from both agents, removing duplicates
let mergeBeliefs (b1: string list) (b2: string list) =
    (b1 @ b2)
    |> List.map (fun l -> l.Trim())
    |> List.filter (fun l -> l.StartsWith("belief "))
    |> Set.ofList |> Set.toList

/// Crossover if belief and quaternion similarity allows
let crossAligned (aPath: string) (bPath: string) (outPath: string) =
    let a = parseTrsx aPath
    let b = parseTrsx bPath
    let qa = encodeAsHurwitz a
    let qb = encodeAsHurwitz b
    let angle = quaternionAngleDeg qa qb

    if angle > 15.0 then
        printfn "⚠ Quaternion angle %.2f too large: skipping crossover." angle
    else
        let beliefs = mergeBeliefs a.Beliefs b.Beliefs
        let lines =
            [   "version: \"offspring_crossed\""
                "description: \"Crossed agent based on quaternion-aligned semantic structure\""
                yield! beliefs
                "quaternion_angle_between_parents: %.2f" |> sprintf angle
                "source: [\"%s\", \"%s\"]" |> sprintf a.Version b.Version
            ]
        File.WriteAllLines(outPath, lines)
        printfn "✅ Created crossover agent at %s with angle %.2f°" outPath angle
```

```
// ==========================================================
// 🧠 grammar_to_ce.fsx — Convert EBNF Grammar to F# CE Format
// ==========================================================

open System
open System.IO
open System.Text.RegularExpressions

let inputGrammarPath = "./.tars/grammars/distilled"
let outputCEPath = "./.tars/grammars/compiled_ce"

let translateRuleToCE (rule: string) : string =
    let ruleName = rule.Split("::=").[0].Trim()
    let production = rule.Split("::=").[1].Trim()
    let tokens = Regex.Split(production, "\s+")

    let ceLines =
        tokens
        |> Array.map (fun token ->
            if token.StartsWith("\"") then $"    do! parseToken {token}"
            elif Regex.IsMatch(token, "<.+>") then $"    do! parseSymbol \"{token.Trim('<', '>')}\""
            else $"    do! parseRaw \"{token}\"")
        |> String.concat "\n"

    $"ce {{
    let! rule = matchRule \"{ruleName}\"
{ceLines}
    return rule
}}"

let processGrammarFile (filePath: string) =
    let lines = File.ReadAllLines(filePath)
    let rules = lines |> Array.filter (fun l -> l.Contains("::="))
    let ceCode = rules |> Array.map translateRuleToCE |> String.concat "\n\n"
    let name = Path.GetFileNameWithoutExtension(filePath)
    let outputPath = Path.Combine(outputCEPath, $"{name}_ce.fsx")
    Directory.CreateDirectory(outputCEPath) |> ignore
    File.WriteAllText(outputPath, ceCode)
    printfn $"Generated CE for: {name}"

// 🔁 Process all grammar files in input folder
Directory.GetFiles(inputGrammarPath, "*.grammar")
|> Array.iter processGrammarFile
```
